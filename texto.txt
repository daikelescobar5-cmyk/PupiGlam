<script>
  // NÚMERO DE WHATSAPP DEL NEGOCIO (¡CAMBIA ESTE NÚMERO POR EL TUYO!)
  // Formato internacional sin '+' ni espacios, solo números. Ejemplo para México: 521234567890
  const BUSINESS_PHONE = "521234567890";   // <--- REEMPLAZA con el número real del negocio (código país + número)
  // Si tu código es +34 600 000 000 → "34600000000" (sin el +)

  // Elementos del formulario
  const form = document.getElementById('orderForm');
  
  form.addEventListener('submit', function(e) {
    e.preventDefault();

    // Obtener valores
    const productType = document.getElementById('productType').value;
    let quantity = document.getElementById('quantity').value;
    if (!quantity || quantity < 1) quantity = 1;
    const customization = document.getElementById('customization').value.trim();
    const customerName = document.getElementById('customerName').value.trim();
    const customerPhone = document.getElementById('customerPhone').value.trim();
    const notes = document.getElementById('notes').value.trim();

    // Validaciones básicas
    if (!customerName) {
      alert("💖 Por favor, ingresa tu nombre para que podamos saludarte.");
      document.getElementById('customerName').focus();
      return;
    }
    if (!customization) {
      alert("🎨 Cuéntanos cómo quieres tu pedido (colores, diseño, detalles). ¡Así lo haremos único!");
      document.getElementById('customization').focus();
      return;
    }

    // Construir mensaje para WhatsApp (con emojis y formato legible)
    let message = `✨ *NUEVO PEDIDO PUPIGLAM* ✨%0A`;
    message += `%0A🧸 *Producto:* ${productType}%0A`;
    message += `🔢 *Cantidad:* ${quantity}%0A`;
    message += `🎨 *Personalización:* ${customization}%0A`;
    message += `👤 *Nombre:* ${customerName}%0A`;
    if (customerPhone) {
      message += `📞 *WhatsApp del cliente:* ${customerPhone}%0A`;
    }
    if (notes) {
      message += `📝 *Notas adicionales:* ${notes}%0A`;
    }
    message += `%0A🌸 *Gracias por confiar en PupiGlam!* Respondemos en breve.`;

    // Codificar mensaje para URL
    const encodedMessage = encodeURIComponent(message);
    
    // Crear link de WhatsApp
    const waLink = `https://wa.me/${BUSINESS_PHONE}?text=${encodedMessage}`;
    
    // Abrir WhatsApp en nueva pestaña
    window.open(waLink, '_blank');
    
    // Opcional: mostrar alerta amigable
    setTimeout(() => {
      alert("✅ ¡Listo! Se abrió WhatsApp con los detalles de tu pedido.\nSolo envía el mensaje y nosotros lo recibiremos al instante. 💬");
    }, 300);
  });

  // Pequeño helper para que los enlaces internos del menú naveguen suavemente
  document.querySelectorAll('.nav-links a, .btn-primary[href="#pedido"]').forEach(anchor => {
    anchor.addEventListener('click', function(e) {
      const href = this.getAttribute('href');
      if (href && href.startsWith('#')) {
        e.preventDefault();
        const targetId = href.substring(1);
        const targetElement = document.getElementById(targetId);
        if (targetElement) {
          targetElement.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }
      }
    });
  });
</script>
